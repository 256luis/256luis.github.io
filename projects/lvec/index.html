lvec.c

A simple dynamic array library for C. It is generic, meaning it will work with any primitive or user-defined type. Like any dynamic array library, users can append, remove, and resize the array at will.

#include "lvec.h"

int main()
{
    int* nums = lvec_new(int);
    if (nums == NULL)
    {
        // handle error
        return 1;
    }

    lvec_append(nums, 10);
    lvec_append(nums, 20);

    // ...

    lvec_free(nums);
}
